def preprocess_article(raw): # 1. HTML 태그 제거 text = strip_html(raw['content']) # 2. 지역 키워드 매칭 district_id = match_district( raw['title'] + text, districts_index # 지역명 역색인 ) # 3. 최소 길이 필터 (200자 미만 skip) if len(text) < 200: return None # 4. url_hash 생성 url_hash = sha256(raw['url']) return { 'district_id': district_id, 'url_hash': url_hash, 'content': text[:3000] # 토큰 절약 }
